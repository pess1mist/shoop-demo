**2. 创建TrendPredictionService**

```java
@Service
public class TrendPredictionService {
    @Autowired
    private TrendPredictionMapper trendPredictionMapper;
    
    public void save(TrendPrediction prediction) {
        trendPredictionMapper.insert(prediction);
    }
    
    public List<TrendPrediction> getHistory(String predictionType, String deptCode) {
        LambdaQueryWrapper<TrendPrediction> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(TrendPrediction::getPredictionType, predictionType);
        if (deptCode != null && !deptCode.isEmpty()) {
            wrapper.eq(TrendPrediction::getDeptCode, deptCode);
        }
        wrapper.orderByDesc(TrendPrediction::getPredictionDate);
        return trendPredictionMapper.selectList(wrapper);
    }
}

public class ExcelExportUtil {
    public static void exportCostAnalysis(HttpServletResponse response, List<Map<String, Object>> data) throws IOException {
        Workbook workbook = new XSSFWorkbook();
        Sheet sheet = workbook.createSheet("成本分析");
        
        // 创建表头
        Row headerRow = sheet.createRow(0);
        String[] headers = {"产品编码", "产品名称", "总成本", "单位成本", "材料费占比"};
        for (int i = 0; i < headers.length; i++) {
            Cell cell = headerRow.createCell(i);
            cell.setCellValue(headers[i]);
        }
        
        // 填充数据
        for (int i = 0; i < data.size(); i++) {
            Row row = sheet.createRow(i + 1);
            Map<String, Object> item = data.get(i);
            row.createCell(0).setCellValue(item.get("productCode").toString());
            row.createCell(1).setCellValue(item.get("productName").toString());
            row.createCell(2).setCellValue(Double.parseDouble(item.get("totalCost").toString()));
            row.createCell(3).setCellValue(Double.parseDouble(item.get("unitCost").toString()));
            row.createCell(4).setCellValue(Double.parseDouble(item.get("materialRatio").toString()));
        }
        
        // 输出到响应
        response.setContentType("application/vnd.openxmlformats-officedocument.spreadsheetml.sheet");
        response.setHeader("Content-Disposition", "attachment; filename=cost_analysis.xlsx");
        workbook.write(response.getOutputStream());
        workbook.close();
    }
}
```
